      int darr[];

module top;
  typedef int pair[2];
  
  function automatic is_intersected(pair q[4]);
      for (int i = 0; i < 4; i++) begin
        int min = (q[i][0] < q[i][1]) ? q[i][0] : q[i][1];
        int max = (q[i][0] > q[i][1]) ? q[i][0] : q[i][1];
        
      	for (int j = 0; j < 4; j++) begin
          if (i != j && ( 
            (min <= q[j][0] && max >= q[j][0]) ||
            (min <= q[j][1] && max >= q[j][1])))
            return 1;
        end
      end
      return 0;
    endfunction

      p = {1, -1};
      q.push_back(p);
